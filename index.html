import java.util.*; public class Main { public static void main(String[] args) { List recipes = new ArrayList<>(); // レシピ追加 recipes.add(new Recipe("無名")); recipes.add(new Recipe( "無名改", 3, 4, new Material("境壊ノ二刀無名", 1), new Material("焔獄魔の逆鱗", 30), new Material("焔獄魔の巌翼", 20), new Material("冰刃魔の逆鱗", 30), new Material("冰刃魔の巌翼", 20), new Material("雷霆魔の逆鱗", 30), new Material("雷霆魔の巌翼", 20), new Material("百足の卵", 50), new Material("百足の毒", 20), new Material("朱の盤の角", 50), new Material("乾いた舌", 20), new Material("斬鋭石", 50), new Material("剛毅石", 50), new Material("霊妙石", 1), new Material("穢珠", 30) )); recipes.add(new Recipe("真打")); recipes.add(new Recipe("修羅")); recipes.add(new Recipe("以蔵武器")); // =============================== // 🔹 入力 Scanner scanner = new Scanner(System.in); System.out.print("レシピ名を入力してください: "); String input = scanner.nextLine(); // =============================== // 🔹 検索 boolean found = false; for (Recipe r : recipes) { if (r.name.equals(input)) { System.out.println("\n【検索結果】"); r.show(); found = true; break; } } // 見つからなかった場合 if (!found) { System.out.println("そのレシピは存在しません"); } } } // =============================== class Recipe { String name; int damage; int attribute; Material[] items; public Recipe(String name, int damage, int attribute, Material... items) { this.name = name; this.damage = damage; this.attribute = attribute; this.items = items; } // 🔹 名前だけ用 public Recipe(String name) { this.name = name; this.damage = 0; this.attribute = 0; this.items = new Material[0]; } public void show() { System.out.println(name + "のステータスは ダメージ:" + damage + " 属性:" + attribute); if (items.length == 0) { System.out.println("必要素材: なし"); return; } System.out.println("必要素材:"); for (Material m : items) { System.out.println(" - " + m.name + " ×" + m.count); } } } // =============================== class Material { String name; int count; public Material(String name, int count) { this.name = name; this.count = count; } }
