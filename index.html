import java.util.*; public class Main { public static void main(String[] args) { List recipes = new ArrayList<>(); // レシピ追加 recipes.add(new Recipe("無名")); recipes.add(new Recipe( "無名改", 3, 4, new Material("境壊ノ二刀無名", 1), new Material("焔獄魔の逆鱗", 30), new Material("焔獄魔の巌翼", 20), new Material("冰刃魔の逆鱗", 30), new Material("冰刃魔の巌翼", 20), new Material("雷霆魔の逆鱗", 30), new Material("雷霆魔の巌翼", 20), new Material("百足の卵", 50), new Material("百足の毒", 20), new Material("朱の盤の角", 50), new Material("乾いた舌", 20), new Material("斬鋭石", 50), new Material("剛毅石", 50), new Material("霊妙石", 1), new Material("穢珠", 30) )); recipes.add(new Recipe("真打")); recipes.add(new Recipe("修羅")); recipes.add(new Recipe("以蔵武器")); Scanner scanner = new Scanner(System.in); // =============================== // 🔹 レシピ名入力 System.out.print("レシピ名を入力してください: "); String input = scanner.nextLine(); Recipe target = null; for (Recipe r : recipes) { if (r.name.equals(input)) { target = r; break; } } if (target == null) { System.out.println("そのレシピは存在しません"); return; } // =============================== // 🔹 本数入力 System.out.print("何本作りますか？: "); int count = scanner.nextInt(); // =============================== // 🔹 結果表示 System.out.println("\n【必要素材（" + count + "本分）】"); if (target.items.length == 0) { System.out.println("素材なし"); return; } for (Material m : target.items) { int total = m.count * count; System.out.println(" - " + m.name + " ×" + total); } } } // =============================== class Recipe { String name; int damage; int attribute; Material[] items; public Recipe(String name, int damage, int attribute, Material... items) { this.name = name; this.damage = damage; this.attribute = attribute; this.items = items; } public Recipe(String name) { this.name = name; this.damage = 0; this.attribute = 0; this.items = new Material[0]; } } // =============================== class Material { String name; int count; public Material(String name, int count) { this.name = name; this.count = count; } }
